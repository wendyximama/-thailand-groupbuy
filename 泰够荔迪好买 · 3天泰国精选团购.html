⚙️ 管理员设置：连接 Google Sheets 自动同步
1. 打开你的 Google Sheets → 点击菜单「扩展程序」→「Apps Script」
2. 把下方代码完整粘贴进去（YOUR_SHEET_ID 已预填你的表格ID）
3. 点击「部署」→「新建部署」→ 类型选「Web 应用」
4. 执行身份选「我」，访问权限选「所有人」→ 点击部署
5. 复制生成的 URL，粘贴到下方保存 ✅ 
// ======================================================
// 泰够荔迪好买 · Google Apps Script 订单同步脚本
// 粘贴到 Apps Script 编辑器，然后部署为 Web 应用
// ======================================================

const SHEET_ID = '1FnwCCXp-q5-akrQq3hghTkSG76GhllDYjXYrUvF1LMc';
const SHEET_NAME = '网站订单'; // 会自动创建此 sheet

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var sheet = ss.getSheetByName(SHEET_NAME);

    // 自动创建 sheet（如不存在）
    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      var headers = ['#','品牌','购买人姓名','联系方式','收货地址',
                     '产品/款式','颜色','尺码','数量','单价(RM)',
                     '小计(RM)','备注','付款状态','付款金额',
                     '付款参考号','发货状态','下单时间'];
      sheet.appendRow(headers);
      sheet.setFrozenRows(1);
      sheet.getRange(1,1,1,headers.length)
           .setBackground('#1C1C2E').setFontColor('#C8963E').setFontWeight('bold');
    }

    var now = Utilities.formatDate(new Date(), 'Asia/Kuala_Lumpur', 'yyyy-MM-dd HH:mm:ss');
    var baseRow = Math.max(sheet.getLastRow(), 1);

    // 每个产品写一行
    data.items.forEach(function(item, idx) {
      sheet.appendRow([
        baseRow + idx,           // #
        item.brand,              // 品牌
        data.name,               // 购买人姓名
        data.phone,              // 联系方式
        data.address,            // 收货地址
        item.product,            // 产品/款式
        item.color  || '',       // 颜色
        item.size   || '',       // 尺码
        item.qty,                // 数量
        item.price,              // 单价(RM)
        item.subtotal,           // 小计(RM)
        data.notes  || '',       // 备注
        '待确认',                 // 付款状态
        data.paymentAmount,      // 付款金额
        data.paymentRef || '',   // 付款参考号
        '待发货',                 // 发货状态
        now                      // 下单时间
      ]);
    });

    // 可选：保存付款截图到 Google Drive（需先创建「团购付款截图」文件夹）
    if (data.receiptBase64 && data.receiptBase64.indexOf(',') > -1) {
      try {
        var folders = DriveApp.getFoldersByName('团购付款截图');
        if (folders.hasNext()) {
          var folder = folders.next();
          var raw = data.receiptBase64.split(',')[1];
          var blob = Utilities.newBlob(
            Utilities.base64Decode(raw), 'image/jpeg',
            data.name + '_' + (data.paymentRef || now) + '.jpg'
          );
          folder.createFile(blob);
        }
      } catch(imgErr) { /* 文件夹不存在则跳过 */ }
    }

    return ContentService
      .createTextOutput(JSON.stringify({ success:true, rows: data.items.length }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ success:false, error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService.createTextOutput('泰够荔迪好买 · 订单同步服务运行中 ✅');
}
